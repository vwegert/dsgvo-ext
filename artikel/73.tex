\chapter{Vorsitz}
\label{ch:73}

\crossrefArticleToReason{73}

\begin{enumerate}

  \item Der Ausschuss wählt aus dem Kreis seiner Mitglieder mit einfacher Mehrheit einen Vorsitzenden und zwei
   stellvertretende Vorsitzende.%
  \label{itm:73-1}

  \item Die Amtszeit des Vorsitzenden und seiner beiden Stellvertreter beträgt fünf Jahre; ihre einmalige Wiederwahl ist
   zulässig.%
  \label{itm:73-2}

\end{enumerate}
